""" Pressure Vessel Calculation Aid Based on ASME BPVC Section VIII Division 1 (reference only) Included: - Cylindrical shell thickness (internal pressure) - Cylindrical shell MAWP - 2:1 Ellipsoidal head thickness - Torispherical head thickness ⚠ DISCLAIMER: This software is a calculation aid only. """ # -------------------------------------------------- # CYLINDRICAL SHELL – INTERNAL PRESSURE (UG-27) # -------------------------------------------------- def cyl_shell_thickness(P, R, S, E, CA=0.0): """ Required minimum shell thickness """ denominator = (S * E) - (0.6 * P) if denominator <= 0: raise ValueError("Invalid inputs: (S × E) must be > 0.6 × P") t = (P * R) / denominator + CA return t def cyl_shell_mawp(t, R, S, E): """ Maximum Allowable Working Pressure (MAWP) """ P = (S * E * t) / (R + 0.6 * t) return P # -------------------------------------------------- # HEAD CALCULATIONS (UG-32 reference) # -------------------------------------------------- def ellipsoidal_head_thickness(P, D, S, E, CA=0.0): """ 2:1 Ellipsoidal Head Thickness """ denominator = (2 * S * E) - (0.2 * P) if denominator <= 0: raise ValueError("Invalid inputs for ellipsoidal head") t = (P * D) / denominator + CA return t def torispherical_head_thickness(P, D, S, E, CA=0.0): """ Torispherical Head Thickness """ denominator = (S * E) - (0.1 * P) if denominator <= 0: raise ValueError("Invalid inputs for torispherical head") t = (0.885 * P * D) / denominator + CA return t # -------------------------------------------------- # EXAMPLE RUN # -------------------------------------------------- if __name__ == "__main__": # Design inputs (units must be consistent) P = 1.6 # Design pressure D = 1000 # Inside diameter (mm) R = D / 2 S = 138 # Allowable stress E = 0.85 # Joint efficiency CA = 1.5 # Corrosion allowance (mm) # Shell t_shell = cyl_shell_thickness(P, R, S, E, CA) mawp_shell = cyl_shell_mawp(t_shell -
